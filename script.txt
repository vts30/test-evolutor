// Jenkins Script Console — run once via Manage Jenkins → Script Console
// Fixes broken layout of perf-report.html in Jenkins artifact viewer.
// Without this, Jenkins CSP blocks Google Fonts and inline styles.
//
// NOTE: resets on Jenkins restart. To persist, add to Jenkins JAVA_OPTS:
//   -Dhudson.model.DirectoryBrowserSupport.CSP="default-src 'self'; style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; font-src 'self' https://fonts.gstatic.com; script-src 'self' 'unsafe-inline'; img-src 'self' data:;"

System.setProperty("hudson.model.DirectoryBrowserSupport.CSP",
    "default-src 'self'; " +
    "style-src 'self' 'unsafe-inline' https://fonts.googleapis.com; " +
    "font-src 'self' https://fonts.gstatic.com; " +
    "script-src 'self' 'unsafe-inline'; " +
    "img-src 'self' data:;"
)

println "CSP updated: " + System.getProperty("hudson.model.DirectoryBrowserSupport.CSP")
